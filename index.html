О себе

Я работаю преподавателем на кафедре фармакологии в Российском университете медицины, веду кружок по научной работе студентов и пишу кандидатскую диссертацию. В своей деятельности объединяю преподавание, исследовательскую практику и управление образовательными проектами. Использую язык R для анализа статистических данных и многого другого (например, эта страница сделана с помощью Quarto)

Уведомление

Создал и поддерживаю Страницу СНК фармакологии — страницу для помощи в научной работе для студентов. На сайте собраны материалы по поиску научной информации, критической оценке источников, оформлению библиографии и участию в научных конкурсах.

required_packages <- c("rorcid", "tibble", "janitor", "dplyr", "knitr")
missing_packages <- required_packages[!sapply(required_packages, requireNamespace, quietly = TRUE)]

if (length(missing_packages) > 0) {
  install.packages(missing_packages)
}

# Загружаем библиотеки
library(tibble)
library(janitor)
library(dplyr)
library(knitr)

# Пытаемся загрузить rorcid и получить данные
tryCatch({
  library(rorcid)
  
  orc_id <- "0000-0002-0254-0516"
  
  # Получаем работы и превращаем в чистую таблицу
  works_df <- rorcid::works(orc_id) %>%
    tibble::as_tibble() %>%
    janitor::clean_names() %>%
    transmute(
      title = title_title_value,
      journal = journal_title_value,
      year = publication_date_year_value
    ) %>%
    mutate(
      year = as.character(year)
    ) %>%
    arrange(desc(year))
  
  works_df %>% 
    knitr::kable(col.names = c("Название работы", "Журнал", "Год"))
  
}, error = function(e) {
  # Если не удалось получить данные из ORCID, показываем сообщение
  cat("**Публикации**\n\n")
  cat("Для просмотра публикаций посетите [ORCID профиль](https://orcid.org/0000-0002-0254-0516)\n\n")
